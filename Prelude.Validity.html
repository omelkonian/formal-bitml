record Validable (A : Type ℓ) : Type (ℓ ⊔ₗ lsuc ℓ′) where
  field Valid : Pred A ℓ′

  Valid? : ⦃ Valid ⁇¹ ⦄ → Decidable¹ Valid
  Valid? _ = dec
