-- Well-rooted traces.

record HasInitial (A : Type ℓ) : Type (1ℓ ⊔ₗ ℓ) where
  field Initial : Pred₀ A

  Initial? : ⦃ Initial ⁇¹ ⦄ → Decidable¹ Initial
  Initial? = dec¹
open HasInitial ⦃...⦄ public

module _ {A : Type ℓ} ⦃ _ : HasInitial A ⦄ (_—↠_ : Rel A ℓ′) where

  record Trace : Type (ℓ ⊔ₗ ℓ′) where
    field
      start end : A
      trace : start —↠ end
      -- .init : Initial start
      init : Initial start
  open Trace public
