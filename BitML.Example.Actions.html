-- secret
_ = ♯▷ ex-ad
-- spend
_ = "x" ▷ˢ ex-ad
-- take branch
_ = "x" ▷ (ex-contracts₄ ‼ # 1)

-- donate
_ = "x" ▷ᵈ B
-- destroy
_ = ⟦ "x₀", "x₁" , "x₂" ⟧ , # 1 ▷ᵈˢ "y"
