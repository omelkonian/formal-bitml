-- Secret lengths.
_≟ₛₗ_ : Decidable {A = Maybe ℕ} _≡_
just x  ≟ₛₗ just y  with x ≟ y
... | no ¬p    = no λ{ refl → ¬p refl }
... | yes refl = yes refl
just _  ≟ₛₗ nothing = no λ ()
nothing ≟ₛₗ just _  = no λ ()
nothing ≟ₛₗ nothing = yes refl

-- Configurations.
_≟ᶜᶠ_ : ∀ {p₁ p₂ p₃} → Decidable {A = Configuration′ p₁ p₂ p₃} _≡_
_∃≟ᶜᶠ_ : Decidable {A = ∃[ p₁ ] ∃[ p₂ ] ∃[ p₃ ] Configuration′ p₁ p₂ p₃} _≡_

∅ᶜ ≟ᶜᶠ ∅ᶜ            = yes refl
∅ᶜ ≟ᶜᶠ (_ auth[ _ ]∶- _) = no λ ()
∅ᶜ ≟ᶜᶠ ⟨ _ ∶ _ ♯ _ ⟩ = no λ ()
∅ᶜ ≟ᶜᶠ (_ ∶ _ ♯ _)   = no λ ()
∅ᶜ ≟ᶜᶠ (_ ∣∣ _ ∶- _) = no λ ()

(` ad) ≟ᶜᶠ (` .ad)          = yes refl
(` ad) ≟ᶜᶠ (p auth[ x ]∶- _)    = no λ ()
(` ad) ≟ᶜᶠ (c′ ∣∣ c′₁ ∶- x) = no λ ()

⟨ c , v ⟩ᶜ ≟ᶜᶠ ⟨ .c , v′ ⟩ᶜ with v ≟ v′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl = yes refl
⟨ c , v ⟩ᶜ ≟ᶜᶠ (p auth[ x ]∶- _)    = no λ ()
⟨ c , v ⟩ᶜ ≟ᶜᶠ (c′ ∣∣ c′₁ ∶- x) = no λ ()

⟨ p , v ⟩ᵈ ≟ᶜᶠ ⟨ .p , .v ⟩ᵈ = yes refl
⟨ p , v ⟩ᵈ ≟ᶜᶠ (p₁ auth[ x ]∶- _) = no λ ()
⟨ p , v ⟩ᵈ ≟ᶜᶠ (c′ ∣∣ c′₁ ∶- x) = no λ ()

(_auth[_]∶-_ {ads} {cs} {vs} {ds} p a _) ≟ᶜᶠ
  (_auth[_]∶-_ {ads′} {cs′} {vs′} {ds′} p′ a′ _)
  with ads SETₐ.≟ₗ ads′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with cs SETᶜ.≟ₗ cs′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with vs SETₙ.≟ₗ vs′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with ds SETₑ.≟ₗ ds′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with p SETₚ.≣ p′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with a ≟ᵃᶜ a′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl = yes refl
(p auth[ x ]∶- _) ≟ᶜᶠ ∅ᶜ = no λ ()
(p auth[ x ]∶- _) ≟ᶜᶠ (` ad) = no λ ()
(p auth[ x ]∶- _) ≟ᶜᶠ ⟨ c , v′ ⟩ᶜ = no λ ()
(p auth[ x ]∶- _) ≟ᶜᶠ ⟨ p₁ , v ⟩ᵈ = no λ ()
(p auth[ x ]∶- _) ≟ᶜᶠ ⟨ x₁ ∶ s ♯ n ⟩ = no λ ()
(p auth[ x ]∶- _) ≟ᶜᶠ (x₁ ∶ s ♯ n) = no λ ()
(p auth[ x ]∶- _) ≟ᶜᶠ (c′ ∣∣ c′₁ ∶- x₁) = no λ ()

⟨ p ∶ s ♯ n ⟩ ≟ᶜᶠ ⟨ p′ ∶ s′ ♯ n′ ⟩
  with p SETₚ.≣ p′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with s SETₛ.≣ s′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with n ≟ₛₗ n′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl = yes refl
⟨ x ∶ s ♯ n ⟩ ≟ᶜᶠ ∅ᶜ = no λ ()
⟨ x ∶ s ♯ n ⟩ ≟ᶜᶠ (p auth[ x₁ ]∶- _) = no λ ()
⟨ x ∶ s ♯ n ⟩ ≟ᶜᶠ (x₁ ∶ s₁ ♯ n₁) = no λ ()
⟨ x ∶ s ♯ n ⟩ ≟ᶜᶠ (c′ ∣∣ c′₁ ∶- x₁) = no λ ()

(p ∶ s ♯ n) ≟ᶜᶠ (p′ ∶ s′ ♯ n′)
  with p SETₚ.≣ p′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with s SETₛ.≣ s′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with n ≟ n′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl = yes refl
(x ∶ s ♯ n) ≟ᶜᶠ ∅ᶜ = no λ ()
(x ∶ s ♯ n) ≟ᶜᶠ (p auth[ x₁ ]∶- _) = no λ ()
(x ∶ s ♯ n) ≟ᶜᶠ ⟨ x₁ ∶ s₁ ♯ n₁ ⟩ = no λ ()
(x ∶ s ♯ n) ≟ᶜᶠ (c′ ∣∣ c′₁ ∶- x₁) = no λ ()

(_∣∣_∶-_ {adsˡ} {radsˡ} {adsʳ} {radsʳ} {ads} {rads}
         {csˡ} {rcsˡ} {csʳ} {rcsʳ} {cs} {rcs}
         {dsˡ} {rdsˡ} {dsʳ} {rdsʳ} {ds} {rds}
         l r _)
  ≟ᶜᶠ
  (_∣∣_∶-_ {adsˡ′} {radsˡ′} {adsʳ′} {radsʳ′} {ads′} {rads′}
         {csˡ′} {rcsˡ′} {csʳ′} {rcsʳ′} {cs′} {rcs′}
         {dsˡ′} {rdsˡ′} {dsʳ′} {rdsʳ′} {ds′} {rds′}
         l′ r′ _)
  with ads SETₐ.≟ₗ ads′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with adsˡ SETₐ.≟ₗ adsˡ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with adsʳ SETₐ.≟ₗ adsʳ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with rads SETₐ.≟ₗ rads′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with radsˡ SETₐ.≟ₗ radsˡ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with radsʳ SETₐ.≟ₗ radsʳ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with cs SETᶜ.≟ₗ cs′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with csˡ SETᶜ.≟ₗ csˡ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with csʳ SETᶜ.≟ₗ csʳ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with rcs SETᶜ.≟ₗ rcs′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with rcsˡ SETᶜ.≟ₗ rcsˡ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with rcsʳ SETᶜ.≟ₗ rcsʳ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with ds SETₑ.≟ₗ ds′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with dsˡ SETₑ.≟ₗ dsˡ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with dsʳ SETₑ.≟ₗ dsʳ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with rds SETₑ.≟ₗ rds′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with rdsˡ SETₑ.≟ₗ rdsˡ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with rdsʳ SETₑ.≟ₗ rdsʳ′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with l ≟ᶜᶠ l′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with r ≟ᶜᶠ r′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl = yes refl
(c ∣∣ c₁ ∶- x) ≟ᶜᶠ ∅ᶜ = no λ ()
(c ∣∣ c₁ ∶- x) ≟ᶜᶠ (` ad) = no λ ()
(c ∣∣ c₁ ∶- x) ≟ᶜᶠ ⟨ c₂ , v′ ⟩ᶜ = no λ ()
(c ∣∣ c₁ ∶- x) ≟ᶜᶠ ⟨ p , v ⟩ᵈ = no λ ()
(c ∣∣ c₁ ∶- x) ≟ᶜᶠ (p auth[ x₁ ]∶- _) = no λ ()
(c ∣∣ c₁ ∶- x) ≟ᶜᶠ ⟨ x₁ ∶ s ♯ n ⟩ = no λ ()
(c ∣∣ c₁ ∶- x) ≟ᶜᶠ (x₁ ∶ s ♯ n) = no λ ()

( (ads  , rads)  , (cs  , rcs)  , (ds  , rds)  , c ) ∃≟ᶜᶠ
  ( (ads′ , rads′) , (cs′ , rcs′) , (ds′ , rds′) , c′ )
  with ads SETₐ.≟ₗ ads′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with rads SETₐ.≟ₗ rads′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with cs SETᶜ.≟ₗ cs′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with rcs SETᶜ.≟ₗ rcs′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with ds SETₑ.≟ₗ ds′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with rds SETₑ.≟ₗ rds′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl
  with c ≟ᶜᶠ c′
... | no ¬p    = no λ{refl →  ¬p refl}
... | yes refl = yes refl
